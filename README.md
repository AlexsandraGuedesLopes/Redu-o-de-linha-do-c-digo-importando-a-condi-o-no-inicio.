# Redu-o-de-linha-do-c-digo-importando-a-condi-o-no-inicio.
Selecionar através do "from" a função especifica na biblioteca (sqrt) no inicio para reduzir linhas no código.
from math import sqrt
n1 = int(input("Digite um número: "))
raiz = sqrt(n1)
print ("A raiz de {} é iqual a {}".format (n1,raiz))
